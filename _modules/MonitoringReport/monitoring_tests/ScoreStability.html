def plot_gini_stability(
    dataframe: pd.DataFrame,
    target: str,
    model_prediction: str,
    date_column_name: str,
    num_bootstraps: int = 1000,
    confidence_level: float = 0.95,
    plot_bar_chart: bool = True,
    custom_bin_edges: List = None,
    custom_bin_labels: List = None,
    label: str = "all",
    time_freq: str = "M",
    savefig_path: str = "./",
) -> Tuple[pd.DataFrame, pd.DataFrame]:
    """
    Plots the stability of the Gini coefficient over time with monthly or quarterly aggregation.

    Args:
        dataframe (pd.DataFrame): The input DataFrame containing the data.
        target (str): The target column name.
        model_prediction (str): The model prediction column name.
        date_column_name (str): The date column name.
        num_bootstraps (int, optional): The number of bootstrap samples for confidence interval calculation. Defaults to 1000.
        confidence_level (float, optional): The confidence level for the bootstrap intervals. Defaults to 0.95.
        plot_bar_chart (bool, optional): Whether to plot a bar chart of the score distribution. Defaults to True.
        custom_bin_edges (List, optional): Custom bin edges for the bar chart. Defaults to None.
        custom_bin_labels (List, optional): Custom bin labels for the bar chart. Defaults to None.
        label (str, optional): A label for the title and filenames. Defaults to "all".
        time_freq (str, optional): Frequency of aggregation ('M' for months, 'Q' for quarters). Defaults to "M".
        savefig_path (str, optional): The path to save the plots. Defaults to "./".

    Returns:
        Tuple[pd.DataFrame, pd.DataFrame]: DataFrames with Gini and mean statistics over time.
    """
    title = f"{str.upper(label)}".replace("_", " ")
    df = dataframe.copy()

    # Convert the date column to datetime format if not already done
    df[date_column_name] = pd.to_datetime(df[date_column_name], errors="coerce")

    if df[date_column_name].isnull().all():
        gini_df = pd.DataFrame(
            {
                "status": ["error"],
                "description": ["Date column is empty"],
            }
        )
        mean_df = gini_df.copy()
        date_s = None
    else:
        # Create a new column with the required date aggregation
        if time_freq == "M":
            df["date_period"] = df[date_column_name].dt.to_period("M")
        elif time_freq == "Q":
            df["date_period"] = df[date_column_name].dt.to_period("Q")
        else:
            raise ValueError(
                "time_freq must be either 'M' for months or 'Q' for quarters"
            )

        date_s = "date_period"

        # Initialize an empty list to store bootstrap results
        gini_df = pd.DataFrame(columns=[date_s, "gini", "lower_bound", "upper_bound"])
        mean_df = pd.DataFrame(
            columns=[
                date_s,
                "mean_target",
                "mean_score",
                "ttl_obj_count",
                "bad_obj_count",
            ]
        )
        month_size = []

        if type(model_prediction) != str:
            df["model_prediction"] = model_prediction
            model_prediction = "model_prediction"

        if type(target) == pd.Series:
            df = pd.concat([df, target], axis=1)
            if type(target) == pd.Series:
                target = target.name

        for period in df[date_s].unique():
            df_period = df[df[date_s] == period]
            bootstrap_ginis = []
            if len(df_period[target].unique()) < 2:
                print(f"{period} is empty or only one value")
                continue
            else:
                month_size.append(len(df_period.index))
                for _ in range(num_bootstraps):
                    boot_sample = resample(
                        df_period, replace=True, n_samples=int(len(df_period) * 0.8)
                    )
                    if len(boot_sample[target].unique()) < 2:
                        continue
                    else:
                        boot_roc_auc = roc_auc_score(
                            boot_sample[target], boot_sample[model_prediction]
                        )
                    boot_gini = 2 * boot_roc_auc - 1
                    bootstrap_ginis.append(boot_gini)

                # Check if bootstrap_ginis is not empty
                if not bootstrap_ginis:
                    print(f"No valid bootstrap samples for {period}")
                    continue

                # Calculate confidence interval
                alpha = (1 - confidence_level) / 2
                lower_bound = np.percentile(bootstrap_ginis, 100 * alpha)
                upper_bound = np.percentile(bootstrap_ginis, 100 * (1 - alpha))

                df_new = pd.DataFrame(
                    [
                        [
                            period,
                            2
                            * roc_auc_score(
                                df_period[target], df_period[model_prediction]
                            )
                            - 1,
                            lower_bound,
                            upper_bound,
                        ]
                    ],
                    columns=gini_df.columns,
                )
                gini_df = pd.concat([df_new, gini_df])

                # Calculate and store mean target and score values
                mean_target = df_period[target].mean()
                mean_score = df_period[model_prediction].mean()
                ttl_obj_count = df_period.shape[0]
                bad_obj_count = df_period[df_period[target] == 1].shape[0]
                mean_df = pd.concat(
                    [
                        pd.DataFrame(
                            [
                                [
                                    period,
                                    mean_target,
                                    mean_score,
                                    ttl_obj_count,
                                    bad_obj_count,
                                ]
                            ],
                            columns=mean_df.columns,
                        ),
                        mean_df,
                    ]
                )

        gini_df = gini_df.sort_values(by=date_s)
        mean_df = mean_df.sort_values(by=date_s)

    # Plot line plot for Gini
    plt.figure(figsize=(12, 6))
    if not df[date_column_name].isnull().all():
        plt.plot(
            gini_df[date_s].astype(str),
            gini_df["gini"],
            marker="o",
            color="blue",
            label="Gini",
        )
        plt.scatter(
            gini_df[date_s].astype(str),
            gini_df["lower_bound"],
            marker="o",
            color="red",
            label="Lower Bound",
        )
        plt.scatter(
            gini_df[date_s].astype(str),
            gini_df["upper_bound"],
            marker="o",
            color="green",
            label="Upper Bound",
        )
        plt.legend(loc="upper center", bbox_to_anchor=(0.5, -0.15), ncol=3)
    else:
        # If data column is empty, plot an empty chart with an annotation
        plt.annotate(
            "Date column is empty",
            xy=(0.5, 0.5),
            xycoords="axes fraction",
            ha="center",
        )
    plt.axhline(y=0, color="black", linewidth=2)
    plt.title(f"Gini Stability. {title}.")
    plt.ylabel("Gini")
    plt.xticks(rotation=45)
    plt.grid(True)
    plt.tight_layout()
    plt.savefig(savefig_path + f"Reports_pics/gini_df_{label}.png")
    plt.show()

    # Plot mean comparison
    plt.figure(figsize=(12, 6))
    if not df[date_column_name].isnull().all():
        plt.plot(
            mean_df[date_s].astype(str),
            mean_df["mean_target"],
            marker="o",
            color="orange",
            linestyle="--",
            label="Mean Target",
        )
        plt.plot(
            mean_df[date_s].astype(str),
            mean_df["mean_score"],
            marker="o",
            color="purple",
            linestyle="--",
            label="Mean Score",
        )
        plt.legend(loc="upper center", bbox_to_anchor=(0.5, -0.15), ncol=3)
    else:
        # If data column is empty, plot an empty chart with an annotation
        plt.annotate(
            "Date column is empty",
            xy=(0.5, 0.5),
            xycoords="axes fraction",
            ha="center",
        )

    plt.axhline(y=0, color="black", linewidth=2)
    plt.title(f"Average target vs predict. {title}.")
    plt.ylabel("Mean Value")
    plt.xticks(rotation=45)
    plt.grid(True)
    plt.tight_layout()
    plt.savefig(savefig_path + f"Reports_pics/mean_df_{label}.png")
    plt.show()

    if plot_bar_chart:
        bar_chart(
            df,
            date_s,
            model_prediction,
            custom_bin_edges=custom_bin_edges,
            custom_bin_labels=custom_bin_labels,
            label=label,
            savefig_path=savefig_path,
        )

    return gini_df, mean_df

def adj_gini(df_mm_test, target, actual_date, score_raw, date_s, ids, df_hist):
    # initialization model NN
    nn_model = NearestNeighbors(n_neighbors=1)

    # Fit model NN
    nn_model.fit(df_hist[[score_raw]])

    # Find NN in actual Data
    distances, indices = nn_model.kneighbors(
        df_mm_test[df_mm_test[date_s] == actual_date][[score_raw]]
    )
    # DF with NN
    neighbors_pairs = pd.DataFrame(
        {
            "client_id_sample1": df_mm_test[df_mm_test[date_s] == actual_date][
                ids
            ].values,
            "client_id_sample2": df_hist.iloc[indices.flatten()][ids].values,
        }
    )

    # Calculate Gini for the original samples
    gini_sample1 = (
        roc_auc_score(
            df_mm_test[df_mm_test[date_s] == actual_date][target],
            df_mm_test[df_mm_test[date_s] == actual_date][score_raw],
        )
        * 2
        - 1
    )

    gini_sample2 = roc_auc_score(df_hist[target], df_hist[score_raw]) * 2 - 1

    # Calculate Gini for the samples from the nearest neighbors
    gini_nn = (
        roc_auc_score(
            df_hist.iloc[indices.flatten()][target],
            df_hist.iloc[indices.flatten()][score_raw],
        )
        * 2
        - 1
    )

    # Display Gini scores
    print(f"Gini for actual date: {gini_sample1}")
    print(f"Gini for Dev: {gini_sample2}")
    print(f"Gini for Dev Nearest Neighbors adj.: {gini_nn}")

def bar_chart(
    df: pd.DataFrame,
    date_col: str,
    score: str,
    custom_bin_edges: List = None,
    custom_bin_labels: List = None,
    label: str = "all",
    savefig_path: str = "./",
) -> pd.DataFrame:
    """
    Plots a bar chart showing the proportion of scores in different bins over time.

    Args:
        df (pd.DataFrame): The input DataFrame.
        date_col (str): The date column name.
        score (str): The score column name.
        custom_bin_edges (List, optional): Custom bin edges for the bar chart. Defaults to None.
        custom_bin_labels (List, optional): Custom bin labels for the bar chart. Defaults to None.
        label (str, optional): A label for the title and filenames. Defaults to "all".
        savefig_path (str, optional): The path to save the plots. Defaults to "./".

    Returns:
        pd.DataFrame: DataFrame with the proportion of scores in each bin over time.
    """
    title = f"{str.upper(label)}".replace("_", " ")
    if date_col:
        unique_dates = df[date_col].unique()
        unique_dates = sorted(unique_dates)

        # Auto bins
        if custom_bin_edges is None:
            bin_labels = ["Low", "Medium_low", "Medium", "Medium_high", "High"]
            df["bin"], bin_edges = pd.qcut(
                df[df[date_col] == unique_dates[0]][score],
                q=5,
                labels=False,
                retbins=True,
                duplicates="drop",
            )
        # Manual bins
        else:
            bin_edges = custom_bin_edges.copy()
            if custom_bin_labels is None:
                bin_labels = []
                for i in range(len(bin_edges) - 1):
                    if bin_edges[i] == 0:
                        bin_labels.append(f"<{bin_edges[i+1]}")
                    elif bin_edges[i + 1] == 1:
                        bin_labels.append(f"{bin_edges[i]}+")
                    else:
                        bin_labels.append(f"{bin_edges[i]}-{bin_edges[i+1]}")
            else:
                bin_labels = custom_bin_labels.copy()
            df["bin"] = pd.cut(
                df[score],
                bins=bin_edges,
                labels=bin_labels,
                include_lowest=True,
                right=False,
            )

        bin_counts = (
            df["bin"]
            .value_counts(normalize=True)
            .reindex(range(len(bin_labels)))
            .fillna(0)
        )
        bin_proportions = pd.DataFrame()
        mean_scores = []  # List to store mean scores for each month

        for date_n in unique_dates:
            df_date = df[
                df[date_col] == date_n
            ].copy()  # Use .copy() to avoid SettingWithCopyWarning
            df_date["bin"] = pd.cut(df_date[score], bins=bin_edges, labels=False)

            bin_counts = (
                df_date["bin"]
                .value_counts(normalize=True)
                .reindex(range(len(bin_labels)))
                .fillna(0)
            )
            bin_proportions[f"{date_n}"] = bin_counts

            # Calculate mean score for each month
            mean_score = df_date[score].mean()
            mean_scores.append(mean_score)

        bin_proportions.T.plot(
            kind="bar", stacked=True, colormap="tab10", figsize=(12, 6)
        )

        plt.legend(loc="upper left", bbox_to_anchor=(1, 1), ncol=1).set_visible(
            True
        )  # Show legend

        plt.ylabel("Proportion")
        plt.xticks(rotation=45)

        # Mean score line
        plt.plot(
            range(len(unique_dates)),
            mean_scores,
            marker="o",
            color="black",
            label="Mean Score PD",
        )
        plt.legend(loc="upper center", bbox_to_anchor=(0.5, -0.15), ncol=3)
        bin_labels_with_mean = ["Mean Score PD"] + bin_labels
        plt.legend(
            bin_labels_with_mean,
            loc="upper center",
            bbox_to_anchor=(0.5, -0.15),
            ncol=3,
        )
    else:
        bin_proportions = pd.DataFrame()
        plt.figure(figsize=(12, 6))
        plt.annotate(
            "Date column is empty",
            xy=(0.5, 0.5),
            xycoords="axes fraction",
            ha="center",
        )

    plt.title(f"Mean score line. {title}.")
    plt.grid(True)
    plt.tight_layout()
    if savefig_path is not None:
        plt.savefig(savefig_path + f"Reports_pics/mean_df_2_{label}.png")
    plt.show()

    return bin_proportions
