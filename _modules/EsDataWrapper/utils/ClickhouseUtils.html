def insert_pd_to_ch(db_client, db_path: str, pd_table: pd.DataFrame) -> int:
    df = pd_table.copy()
    # infer types
    for col in df.columns:
        col_type = df[col].dtype
        name = col_type.name
        if col_type != object and name != "category" and name != "string":
            c_min = df[col].min()
            c_max = df[col].max()
            if pd.isnull(c_min):
                df = df.drop(labels=col, axis=1)
                log.warning("Column %s contains invalid values - skipping from upload")
                continue
            if str(col_type).lower().startswith("int") or str(
                col_type
            ).lower().startswith("uint"):
                if c_min > np.iinfo(np.int8).min and c_max < np.iinfo(np.int8).max:
                    df[col] = df[col].astype("Int8")
                elif c_min > np.iinfo(np.int16).min and c_max < np.iinfo(np.int16).max:
                    df[col] = df[col].astype("Int16")
                elif c_min > np.iinfo(np.int32).min and c_max < np.iinfo(np.int32).max:
                    df[col] = df[col].astype("Int32")
                else:
                    df[col] = df[col].astype("Int64")
            elif str(col_type).startswith("float"):
                if (
                    c_min > np.finfo(np.float32).min
                    and c_max < np.finfo(np.float32).max
                ):
                    df[col] = df[col].astype(np.float32)
                else:
                    df[col] = df[col].astype(np.float64)
            elif str(col_type).startswith("date"):
                continue
            else:
                log.error("Unsupported type %s" % str(col_type))
                return 1
        else:
            df[col] = df[col].astype("string")
    # remove last comma
    log.info("Inserting data into %s" % db_path)
    db_client.insert_df(db_path, df)
    return 0

def upload_pd_to_ch(
    db_client, db_path: str, pd_table: pd.DataFrame, primary_key: str = "borrower_id"
) -> int:
    """
    Uploads a Pandas DataFrame to a ClickHouse database.

    Args:
        db_client: The ClickHouse client used to execute commands and insert data.
        db_path (str): The ClickHouse table path where the DataFrame will be uploaded.
        pd_table (pd.DataFrame): The Pandas DataFrame to be uploaded.
        primary_key (str): The primary key for the ClickHouse table. Defaults to "borrower_id".

    Returns:
        int: Returns 0 if the upload is successful, or 1 if an error occurs.

    Raises:
        ValueError: If the DataFrame contains unsupported data types.
        RuntimeError: If there is an issue with creating the table or inserting data into ClickHouse.
    """
    db_client.command(f"DROP TABLE IF EXISTS {db_path}")
    df = pd_table.copy()
    # infer types
    create_query = f"CREATE TABLE {db_path} ("
    for col in df.columns:
        col_type = df[col].dtype
        name = col_type.name
        if col_type != object and name != "category" and name != "string":
            c_min = df[col].min()
            c_max = df[col].max()
            if pd.isnull(c_min):
                df = df.drop(labels=col, axis=1)
                continue
            if str(col_type).lower().startswith("int") or str(
                col_type
            ).lower().startswith("uint"):
                if c_min > np.iinfo(np.int8).min and c_max < np.iinfo(np.int8).max:
                    df[col] = df[col].astype("Int8")
                    typing = (
                        "Nullable(Int8)" if df[col].isnull().values.any() else "Int8"
                    )
                    create_query += f"{col} {typing}, "
                elif c_min > np.iinfo(np.int16).min and c_max < np.iinfo(np.int16).max:
                    df[col] = df[col].astype("Int16")
                    typing = (
                        "Nullable(Int16)" if df[col].isnull().values.any() else "Int16"
                    )
                    create_query += f"{col} {typing}, "
                elif c_min > np.iinfo(np.int32).min and c_max < np.iinfo(np.int32).max:
                    df[col] = df[col].astype("Int32")
                    typing = (
                        "Nullable(Int32)" if df[col].isnull().values.any() else "Int32"
                    )
                    create_query += f"{col} {typing}, "
                else:
                    df[col] = df[col].astype("Int64")
                    typing = (
                        "Nullable(Int64)" if df[col].isnull().values.any() else "Int64"
                    )
                    create_query += f"{col} {typing}, "
            elif str(col_type).startswith("float"):
                # if c_min > np.finfo(np.float16).min and c_max < np.finfo(np.float16).max:
                #     df[col] = df[col].astype(np.float16)
                #     typing = 'Nullable(Float16)' if df[col].isnull().values.any() else 'Float16'
                #     create_query += f"{col} {typing}, "
                if (
                    c_min > np.finfo(np.float32).min
                    and c_max < np.finfo(np.float32).max
                ):
                    df[col] = df[col].astype(np.float32)
                    typing = (
                        "Nullable(Float32)"
                        if df[col].isnull().values.any()
                        else "Float32"
                    )
                    create_query += f"{col} {typing}, "
                else:
                    df[col] = df[col].astype(np.float64)
                    typing = (
                        "Nullable(Float64)"
                        if df[col].isnull().values.any()
                        else "Float64"
                    )
                    create_query += f"{col} {typing}, "
            elif str(col_type).startswith("date"):
                typing = (
                    "Nullable(DateTime)"
                    if df[col].isnull().values.any()
                    else "DateTime"
                )
                create_query += f"{col} {typing}, "
            elif str(col_type).lower().startswith("bool"):
                typing = "Nullable(Bool)" if df[col].isnull().values.any() else "Bool"
                create_query += f"{col} {typing}, "
            else:
                log.error("Unsupported type %s" % str(col_type))
                return 1
        else:
            df[col] = df[col].astype("string")
            typing = "Nullable(String)" if df[col].isnull().values.any() else "String"
            create_query += f"{col} {typing}, "
    # remove last comma
    create_query = create_query[:-2]
    log.info("Creating new empty table %s" % db_path)
    create_query += f") ENGINE = MergeTree PRIMARY KEY {primary_key}"
    # print(create_query)
    db_client.command(create_query)
    log.info("Inserting data into %s" % db_path)
    db_client.insert_df(db_path, df)
    return 0
