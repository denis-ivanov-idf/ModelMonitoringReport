def collect_target_scoring(
    db_client,
    model_name: str,
    unnax: str = None,
) -> pd.DataFrame:
    """
    Collects target from the database based on the specified model name.

    Args:
        db_client: The ClickHouse client used to execute commands and fetch data.
        model_name (str): The name of the model for which target is being collected.
        unnax (str): The unnax flag (MM or PLAZO) for model.

    Returns:
        pd.DataFrame: A DataFrame containing the target.

    Raises:
        ValueError: If the model name is not recognized.
    """
    if model_name == "sm_Plz_mm2plazo":
        query = """
                select borrower_id_mm                                      as mm_borrower_id
                    , borrower_id_crdes                                    as crdes_borrower_id
                    , date_of_application                                  as plz_date_application
                    , date_issuance                                        as plz_date_issuance
                    , og_approve_flg                                       as open_gate_flag
                    , RiskSegment_simulation                               as risk_segment
                    , last_credit_number_mm_all                            as mm_credit_number
                    , if(toDate(date_issuance) < today() - interval 6 month - interval 2 day,
                        if(dr.expired_days_count_mob_6 > 30, 1, 0), Null) as target30m6
                    , if(toDate(date_issuance) < today() - interval 3 month - interval 2 day,
                        if(dr.expired_days_count_mob_3 > 30, 1, 0), Null) as target30m3
                from product_analytics.risk_holybook rh
                where borrower_id_mm > 0
                AND toDate(date_issuance) >= '2022-07-01'
                """
        df = db_client.query_df(query)
    elif model_name == "sm_Plz_BehUnnax":
        if unnax == "MM":
            query = """
                    select coalesce(borrower_id_mm, 0)                                           as mm_borrower_id
                        , borrower_id_crdes                                                     as crdes_borrower_id
                        , date_issuance                                                         as plz_date_issuance
                        , og_approve_flg                                                        as open_gate_flag
                        , RiskSegment_simulation                                                as risk_segment
                        , last_credit_number_mm_all                                             as mm_credit_number
                        , if(toDate(date_issuance) < today() - interval 6 month - interval 2 day,
                                            if(dr.expired_days_count_mob_6 > 30, 1, 0), Null) as target30m6
                        , if(toDate(date_issuance) < today() - interval 3 month - interval 2 day,
                                            if(dr.expired_days_count_mob_3 > 30, 1, 0), Null) as target30m3
                    from product_analytics.risk_holybook rh
                    where toDate(date_issuance) >= '2022-11-01'
                    AND borrower_id_crdes not in
                        (select borrower_id
                        from escc_master.unnax_parsed_response
                        where request_code != ''
                            AND communication_type = 'FITNANCE_READ')
                    """
        elif unnax == "PLAZO":
            query = """
                    select coalesce(borrower_id_mm, 0)               as mm_borrower_id
                        , borrower_id_crdes                          as crdes_borrower_id
                        , date_issuance                              as plz_date_issuance
                        , og_approve_flg                             as open_gate_flag
                        , RiskSegment_simulation                     as risk_segment
                        , last_credit_number_mm_all                  as mm_credit_number
                        , if(toDate(date_issuance) < today() - interval 6 month - interval 2 day,
                                            if(dr.expired_days_count_mob_6 > 30, 1, 0), Null) as target30m6
                        , if(toDate(date_issuance) < today() - interval 3 month - interval 2 day,
                                            if(dr.expired_days_count_mob_3 > 30, 1, 0), Null) as target30m3
                    from product_analytics.risk_holybook rh
                    where toDate(date_issuance) >= '2022-11-01'
                    AND borrower_id_crdes in
                        (select borrower_id
                        from escc_master.unnax_parsed_response
                        where request_code != ''
                            AND communication_type = 'FITNANCE_READ')
                    """
        df = db_client.query_df(query)
        df["mm_borrower_id"] = df["mm_borrower_id"].fillna(0).astype(int)
    return df

def collect_data_scoring(
    db_client, model_name: str, base_table: str, unnax: str = None
) -> pd.DataFrame:
    """
    Collects data for scoring based on the specified model name.

    Args:
        db_client: The ClickHouse client used to execute commands and fetch data.
        model_name (str): The name of the model for which data is being collected.
        unnax (str): The unnax flag (MM or PLAZO) for model.
    Returns:
        pd.DataFrame: A DataFrame containing the collected data.

    Raises:
        ValueError: If the model name is not recognized.
    """
    if model_name == "sm_Plz_mm2plazo":
        query = f"""
        with sc as (select T1.mm_borrower_id       as mm_borrower_id
                        , crdes_borrower_id
                        , open_gate_flag
                        , risk_segment
                        , mm_credit_number
                        , T1.plz_date_application as plz_date_application
                        , T1.plz_date_issuance as plz_date_issuance
                        , T2.penalty_amount       as penalty_amount
                        , api_ping_cnt_2m
                        , ltb_t.ltb_sum_points    as ltb_sum_points
                        , target30m6
                        , target30m3
                    from {base_table} T1
                            left join (select borrower_id
                                            , plz_date_application
                                            , arrayFilter((var) -> (var > 0 and var is not null),
                                                        groupArray(penalty_amount))[1] as penalty_amount
                                        from (
                                                select T2.borrower_id as borrower_id
                                                    , plz_date_application
                                                    , penalty_amount

                                                from {base_table} T1
                                                        join es_report.inflow T2 on T1.mm_borrower_id = T2.borrower_id
                                                where T2.status = 'COMPLETED'
                                                and T2.date_repaid < T1.plz_date_application
                                                order by mm_borrower_id, date_requested desc
                                                ) tmp
                                        group by borrower_id, plz_date_application
                        ) T2 on T2.borrower_id = T1.mm_borrower_id
                        and T2.plz_date_application = T1.plz_date_application
                            left join (select mm_borrower_id
                                            , plz_date_application
                                            , length(ping_ids)           as api_ping_cnt_total
                                            , arrayCount((d) -> (toDate(d) >= toDate(plz_date_application) - interval 2 month),
                                                        creation_dates) as api_ping_cnt_2m
                                        from (select b.id                         as mm_borrower_id
                                                , T1.plz_date_application      as plz_date_application

                                                , groupArray(pi.id)            as ping_ids
                                                , groupArray(pi.creation_date) as creation_dates
                                                , groupArray(pi.stage)         as stages
                                                , groupArray(pi.status)        as statuses
                                                , groupArray(partner.name)     as partner_names
                                                , groupArray(partner.type)     as partner_types
                                                , groupArray(partner.status)   as parnter_statuses
                                                , topK(2)(partner.name)        as top2_partners
                                            from {base_table} T1
                                                    join es_debezium.borrower as b
                                                        on T1.mm_borrower_id = b.id
                                                    join es_debezium.personal_data as pd
                                                        on b.personal_data_id = pd.id
                                                    join es_debezium.partner_integration as pi
                                                        on pi.passport_identification_number_sip_hash_128 =
                                                                pd.passport_identification_number_sip_hash_128
                                                    join es_debezium.partner as partner
                                                        on pi.partner_id = partner.id
                                            where pi.creation_date < T1.plz_date_application
                                            group by b.id, T1.plz_date_application
                                                )
                        ) as ping_t on ping_t.mm_borrower_id = T1.mm_borrower_id
                        and ping_t.plz_date_application = T1.plz_date_application
                            left join (select T1.mm_borrower_id as mm_borrower_id
                                            , plz_date_application
                                            , sum(ltb.points)   as ltb_sum_points
                                            , count(ltb.points) as ltb_cnt_points
                                        from {base_table} T1
                                                join es_debezium.life_time_bonus as ltb on ltb.borrower_id = T1.mm_borrower_id
                                        where ltb.date < T1.plz_date_application
                                        group by mm_borrower_id, plz_date_application
                        ) as ltb_t on ltb_t.mm_borrower_id = T1.mm_borrower_id
                        and ltb_t.plz_date_application = T1.plz_date_application)
        select crdes_borrower_id                                          as borrower_id,
            plz_date_issuance,
            open_gate_flag,
            risk_segment,
            mm_credit_number,
            target30m6,
            target30m3,
            penalty_amount,
            api_ping_cnt_2m,
            ltb_sum_points,
            case
                when penalty_amount is null then 0.1041768666338863
                when penalty_amount < 38.03620910644532 then 0.1041768666338863
                when penalty_amount >= 38.03620910644532 then -0.5066542294069872
                end                                                    as WOE_penalty_amount,

            case
                when api_ping_cnt_2m is null then 0.4590941550216412
                when api_ping_cnt_2m < 1e-05 then 0.4590941550216412
                when api_ping_cnt_2m >= 1e-05 and api_ping_cnt_2m < 5.500000000000001 then -0.14917098689265548
                when api_ping_cnt_2m >= 5.500000000000001 then -1.003921192542504
                end                                                    as WOE_api_ping_cnt_2m,

            case
                when ltb_sum_points is null then 0.46776652696277493
                when ltb_sum_points < 535.5000000000001 then -0.7769820921055327
                when ltb_sum_points >= 535.5000000000001 and ltb_sum_points < 4294.000000000001 then -0.2131355424186427
                when ltb_sum_points >= 4294.000000000001 then 0.46776652696277493
                end                                                    as WOE_ltb_sum_points
        from sc
        """
        df = db_client.query_df(query)
    elif model_name == "sm_Plz_BehUnnax":
        if unnax == "MM":
            query = f"""
            SELECT crdes_borrower_id as borrower_id
                , plz_date_issuance
                , upload_date
                , open_gate_flag
                , risk_segment
                , mm_credit_number

                , target30m3
                , share_70perc_balance_occur
                , cnt_less_neg250eu_am_trans_l3m

                , less_10perc_balance_occur_l3m
                , loans_income_l1m_new
                , CASE
                    WHEN median_balance_amount_l3m <= 148.84 THEN -0.21126516369158121
                    WHEN median_balance_amount_l3m > 148.84 THEN 0.10406061306564474
                    WHEN median_balance_amount_l3m IS NULL THEN 0.0
                    ELSE 0 END AS median_balance_amount_l3m_WOE
                , CASE
                    WHEN cnt_less_neg250eu_am_trans_l3m <= 13.5 THEN 0.13014780260676284
                    WHEN cnt_less_neg250eu_am_trans_l3m > 13.5 THEN -0.20205859029893514
                    WHEN cnt_less_neg250eu_am_trans_l3m IS NULL THEN 0.0
                    ELSE 0 END AS cnt_less_neg250eu_am_trans_l3m_WOE
                , CASE
                    WHEN loans_expenses_l6m <= -3.0 THEN 0.3103099303584578
                    WHEN loans_expenses_l6m > -3.0 THEN -0.47405379316231144
                    WHEN loans_expenses_l6m IS NULL THEN 0.0
                    ELSE 0 END AS loans_expenses_l6m_WOE
                , CASE
                    WHEN loan_amount_l1m_to_prev_l1m <= 0.45 THEN -0.46599284490067927
                    WHEN loan_amount_l1m_to_prev_l1m > 0.45 THEN 0.2330001300608806
                    WHEN loan_amount_l1m_to_prev_l1m IS NULL THEN 0.0
                    ELSE 0 END AS loan_amount_l1m_to_prev_l1m_WOE
                , CASE
                    WHEN neg_loan_to_avg_balance_l3m <= -1.07 THEN 0.13376674153036006
                    WHEN neg_loan_to_avg_balance_l3m > -1.07 THEN -0.4274820366004374
                    WHEN neg_loan_to_avg_balance_l3m IS NULL THEN 0.0
                    ELSE 0 END AS neg_loan_to_avg_balance_l3m_WOE
                , CASE
                    WHEN less_10perc_balance_occur_l3m <= 41.5 THEN 0.34798774662033694
                    WHEN less_10perc_balance_occur_l3m > 41.5 THEN -0.23187869584642007
                    WHEN less_10perc_balance_occur_l3m IS NULL THEN 0.0
                    ELSE 0 END AS less_10perc_balance_occur_l3m_WOE
                , CASE
                    WHEN share_70perc_balance_occur <= 0.51 THEN 0.1305168199178619
                    WHEN share_70perc_balance_occur > 0.51 THEN -0.15625429824637327
                    WHEN share_70perc_balance_occur IS NULL THEN 0.0
                    ELSE 0 END AS share_70perc_balance_occur_WOE
                , CASE
                    WHEN loans_income_l1m_new <= 1798.0 THEN 0.10654244380240496
                    WHEN loans_income_l1m_new > 1798.0 THEN -0.29992503205123155
                    WHEN loans_income_l1m_new IS NULL THEN 0.0
                    ELSE 0 END AS loans_income_l1m_new_WOE
            FROM (SELECT T1.crdes_borrower_id                                 as crdes_borrower_id
                    , T1.target30m6                                           as target30m6
                    , T1.target30m3                                           as target30m3
                    , T1.mm_borrower_id                                       as mm_borrower_id
                    , T1.plz_date_issuance                                    as plz_date_issuance
                    , T1.open_gate_flag                                       as open_gate_flag
                    , T1.risk_segment                                         as risk_segment
                    , T1.mm_credit_number                                     as mm_credit_number
                    , T2.upload_date                                          as upload_date
                    , round(arraySum(arrayFilter(
                        (sa, sct, svd) -> (sct in ('FAST LOANS', 'CREDIT LINE', 'CONSUMER LOANS', 'GENERAL LOANS',
                                                'CREDIT CARD PAYMENT & TRANSFERS', 'CAR LEASING',
                                                'MORTGAGE') and
                                        dateDiff('day', toDate(svd), toDate(upload_date)) <= 60 and
                                        dateDiff('day', toDate(svd), toDate(upload_date)) > 30 and
                                        sa < 0 and toDate(svd) < toDate(T1.plz_date_issuance)),
                        statements_amounts,
                        statements_category_titles,
                        statements_value_dates)),
                            2)                                                as unnax_negative_loan_amount_prev_l1m
                    , round(arraySum(arrayFilter(
                        (sa, sct, svd) -> (sct in ('FAST LOANS', 'CREDIT LINE', 'CONSUMER LOANS', 'GENERAL LOANS',
                                                'CREDIT CARD PAYMENT & TRANSFERS', 'CAR LEASING',
                                                'MORTGAGE') and
                                        dateDiff('day', toDate(svd), toDate(upload_date)) <= 30 and
                                        sa < 0 and toDate(svd) < toDate(T1.plz_date_issuance)),
                        statements_amounts,
                        statements_category_titles,
                        statements_value_dates)),
                            2)                                                as unnax_negative_loan_amount_l1m
                    , round(if(unnax_negative_loan_amount_prev_l1m != 0 and unnax_negative_loan_amount_l1m != 0,
                                unnax_negative_loan_amount_l1m / unnax_negative_loan_amount_prev_l1m, -1),
                            2)                                                as loan_amount_l1m_to_prev_l1m
                    , round(arrayReduce('median', arrayFilter(
                        (balance, month) -> (dateDiff('month', month, mnth_upload_date) <= 3 and balance >= 0 and
                                            toDate(month) <= toDate(T1.plz_date_issuance)),
                        statements_balances,
                        mnth_transactions)),
                            2)                                                as median_balance_amount_l3m
                    , arrayCount((sa, svd) -> (sa < -250 and dateDiff('day', toDate(svd), toDate(upload_date)) <= 90 and
                                                toDate(svd) < toDate(T1.plz_date_issuance)),
                                    statements_amounts,
                                    statements_value_dates)                      as cnt_less_neg250eu_am_trans_l3m
                    , arrayCount((sb, svd) -> (sb <= 0.1 * avg_balance_amount_l3m
                    and dateDiff('day', toDate(svd), toDate(upload_date)) <= 90 and toDate(svd) <= toDate(T1.plz_date_issuance)),
                                    statements_balances,
                                    statements_value_dates)                      as less_10perc_balance_occur_l3m
                    , round(arrayAvg(arrayFilter(
                        (balance, month) -> (dateDiff('month', month, mnth_upload_date) <= 3 and balance >= 0 and
                                            toDate(month) <= toDate(T1.plz_date_issuance)),
                        statements_balances,
                        mnth_transactions)),
                            2)                                                as avg_balance_amount_l3m
                    , round(arrayReduce('sum',
                                        arrayFilter(
                                                (sa, sct, svd) -> (sct in ('FAST LOANS', 'CREDIT LINE', 'CONSUMER LOANS',
                                                                            'GENERAL LOANS',
                                                                            'CREDIT CARD PAYMENT & TRANSFERS', 'CAR LEASING',
                                                                            'MORTGAGE') and
                                                                    dateDiff('day', toDate(svd), toDate(upload_date)) <= 90 and
                                                                    sa < 0 and toDate(svd) < toDate(T1.plz_date_issuance)),
                                                statements_amounts,
                                                statements_category_titles,
                                                statements_value_dates)),
                            2)                                                as unnax_negative_loan_amount_l3m
                    , round(if(avg_balance_amount_l3m != 0,
                                unnax_negative_loan_amount_l3m / avg_balance_amount_l3m, -1),
                            2)                                                as neg_loan_to_avg_balance_l3m
                    , round(arraySum(arrayFilter((sa, svd, sct) -> (dateDiff('day', svd, upload_date) <= 180
                    and sct in ('GENERAL LOANS', 'FAST LOANS', 'CONSUMER LOANS') and toDate(svd) < toDate(T1.plz_date_issuance)),
                                                    statements_amounts,
                                                    statements_value_dates,
                                                    statements_category_titles)),
                            2)                                                as loans_expenses_l6m
                    , round(arraySum(arrayFilter((sa, svd, sct) -> (dateDiff('month', svd, upload_date) <= 1
                    and sct in
                        ('GENERAL LOANS', 'FAST LOANS', 'CONSUMER LOANS', 'CREDIT LINE', 'CREDIT CARD PAYMENT & TRANSFERS') and
                                                                    toDate(svd) < toDate(T1.plz_date_issuance)
                    and sa > 0),
                                                    statements_amounts,
                                                    statements_value_dates,
                                                    statements_category_titles)),
                            2)                                                as loans_income_l1m_new
                    , arrayCount((sb, svd) -> (sb <= 0.7 * avg_balance_all
                    and toDate(svd) <= toDate(upload_date) and toDate(svd) <= toDate(T1.plz_date_issuance)),
                                    statements_balances,
                                    statements_value_dates)                      as less_70perc_balance_occur

                    , round(arrayAvg(arrayFilter((balance, svd) -> (toDate(svd) < toDate(T1.plz_date_issuance)),
                                                    statements_balances,
                                                    statements_value_dates)), 2) as avg_balance_all
                    , round(if(less_70perc_balance_occur = 0, 1, less_70perc_balance_occur) /
                            if(avg_balance_all = 0, 1, avg_balance_all),
                            2)                                                as share_70perc_balance_occur
                FROM {base_table} T1

                                            T1.plz_date_issuance                  as plz_date_issuance,
                                            max(T.mnth_upload_date)               as mnth_upload_date,

                                                , statements_concepts
                                            WHERE 1 = 1
                                                AND statements_value_date >= '2021-01-01'
                                                AND toStartOfDay(statements_value_date) < toStartOfDay(upload_date)
                                            ) T
                                            JOIN es_debezium.cp_unnax cu on T.request_code = cu.request_code
                                            JOIN es_debezium.credit i on cu.credit_request_id = i.id
                                            JOIN {base_table} T1
                                                    on T1.mm_borrower_id = i.borrower_id
                                            JOIN (
                                        SELECT *
                                        FROM (
                                                SELECT i.borrower_id         as   mm_borrower_id
                                                        , T1.request_code    as  request_code
                                                        , mnth_upload_date
                                                        , mnth_transaction
                                                        , ROW_NUMBER() OVER (PARTITION BY i.borrower_id, mnth_transaction ORDER BY mnth_upload_date DESC) RN
                                                FROM (
                                                            SELECT request_code
                                                                , toStartOfMonth(upload_date)           mnth_upload_date
                                                                , toStartOfMonth(statements_value_date) mnth_transaction
                                                            FROM es_master.unnax_parsed_response
                                                                    ARRAY JOIN statements_value_date
                                                            WHERE statements_value_date >= '2021-01-01'
                                                            AND statements_value_date < today()
                                                            ) T1
                                                            JOIN es_debezium.cp_unnax cu on T1.request_code = cu.request_code
                                                            JOIN es_report.inflow i on cu.credit_request_id = i.id
                                                WHERE cu.valid_report = 1
                                                GROUP BY i.borrower_id, T1.request_code, mnth_upload_date, mnth_transaction) T2
                                        WHERE RN = 1
                                        ) ts
                                                    ON i.borrower_id =
                                                    ts.mm_borrower_id
                                                        AND ts.request_code = T.request_code
                                                        AND ts.mnth_upload_date = T.mnth_upload_date
                                                        AND ts.mnth_transaction = T.mnth_transaction
                                                        AND toDate(T.upload_date) <= toDate(T1.plz_date_issuance)
                                    GROUP BY i.borrower_id, T1.plz_date_issuance
                    ) T2 ON T1.mm_borrower_id = T2.mm_borrower_id
                    and T1.plz_date_issuance = T2.plz_date_issuance
                    ) as sc
                """
        elif unnax == "PLAZO":
            query = f"""
            SELECT crdes_borrower_id as borrower_id
                , plz_date_issuance
                , upload_date
                , open_gate_flag
                , risk_segment
                , mm_credit_number
                , target30m6
                , target30m3
                , share_70perc_balance_occur

                    WHEN median_balance_amount_l3m > 148.84 THEN 0.10406061306564474
                    WHEN median_balance_amount_l3m IS NULL THEN 0.0
                    ELSE 0 END                                           AS median_balance_amount_l3m_WOE

                    WHEN cnt_less_neg250eu_am_trans_l3m <= 13.5 THEN 0.13014780260676284
                    WHEN cnt_less_neg250eu_am_trans_l3m > 13.5 THEN -0.20205859029893514
                    WHEN cnt_less_neg250eu_am_trans_l3m IS NULL THEN 0.0
