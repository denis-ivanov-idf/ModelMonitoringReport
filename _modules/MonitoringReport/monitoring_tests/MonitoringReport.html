def excel_saver(
    gini_df: pd.DataFrame,
    psi_table: pd.DataFrame,
    mean_df: pd.DataFrame,
    feature_type_check: pd.DataFrame,
    name: str = "model_name",
    label: str = "all",
    mode: str = "w",
    resource_path: str = "./",
) -> None:
    """
    Saves multiple DataFrames to an Excel file, each in a separate sheet, and inserts associated images.

    Args:
        gini_df (pd.DataFrame): DataFrame containing Gini coefficient data.
        psi_table (pd.DataFrame): DataFrame containing PSI (Population Stability Index) data.
        mean_df (pd.DataFrame): DataFrame containing mean values.
        feature_type_check (pd.DataFrame): DataFrame containing feature type check data.
        name (str, optional): Name of the Excel file (without extension). Defaults to "model_name".
        label (str, optional): Label for the sheet names and image filenames. Defaults to "all".
        mode (str, optional): Mode for opening the Excel file ('w' for write, 'a' for append). Defaults to "w".
        resource_path (str, optional): Base path for saving the Excel file and images. Defaults to "./".

    Returns:
        None
    """
    path = resource_path + "Reports_xls/" + name + ".xlsx"
    log.info("Creating Excel report in %s" % path)
    with pd.ExcelWriter(path, engine="openpyxl", mode=mode) as writer:
        gini_df.to_excel(writer, index=True, sheet_name=f"gini_df_{label}")
        mean_df.to_excel(writer, index=True, sheet_name=f"mean_df_{label}")
        psi_table.to_excel(writer, index=True, sheet_name=f"psi_table_{label}")
        feature_type_check.to_excel(writer, index=True, sheet_name=f"nan_check_{label}")

        wb = writer.book

        resource_path += "/Reports_pics/"
        sheets_images = [
            (f"gini_df_{label}", resource_path + f"gini_df_{label}.png"),
            (f"mean_df_{label}", resource_path + f"mean_df_{label}.png"),
            (
                f"nan_check_{label}",
                resource_path + f"nan_check_{label}.png",
            ),
        ]
        for sheet_name, img_path in sheets_images:
            ws = wb[sheet_name]
            img = Image(img_path)
            img.width, img.height = 600, 300
            ws.add_image(img, "I1")
